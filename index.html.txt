        <p id="feedback-text"></p>
        <button class="btn" onclick="startGame()">حاولي مرة أخرى</button>

    <script>
        const canvas = document.getElementById('gameCanvas');
        const ctx = canvas.getContext('2d');
        const scoreEl = document.getElementById('score');
        const finalScoreEl = document.getElementById('final-score');
        const feedbackEl = document.getElementById('feedback-text');
        const startScreen = document.getElementById('start-screen');
        const gameOverScreen = document.getElementById('game-over-screen');

        let animationId;
        let score = 0;
        let gameRunning = false;
        let frame = 0;
        let items = [];

        const itemTypes = [
            { emoji: "🗑️", points: 10 },
            { emoji: "🧃", points: 15 },
            { emoji: "💧", points: 20 },
            { emoji: "🧹", points: 5 }
        ];

        function resize() {
            canvas.width = window.innerWidth;
            canvas.height = window.innerHeight;
        }
        window.addEventListener('resize', resize);
        resize();

        const player = {
            x: canvas.width / 2,
            y: canvas.height - 100,
            width: 70,
            emoji: "👧🏻"
        };

        let targetX = canvas.width / 2;

        canvas.addEventListener('touchmove', (e) => {
            e.preventDefault();
            targetX = e.touches[0].clientX;
        }, { passive: false });

        canvas.addEventListener('mousemove', (e) => {
            targetX = e.clientX;
        });

        function spawn() {
            if (frame % 40 === 0) {
                const type = itemTypes[Math.floor(Math.random() * itemTypes.length)];
                items.push({
                    x: Math.random() * (canvas.width - 40) + 20,
                    y: -50,
                    speed: 4 + Math.random() * 3,
                    ...type
                });
            }
        }

        function update() {
            player.x += (targetX - player.x) * 0.15;

            for (let i = items.length - 1; i >= 0; i--) {
                items[i].y += items[i].speed;

                // Collision
                const dx = player.x - items[i].x;
                const dy = (canvas.height - 100) - items[i].y;
                const dist = Math.sqrt(dx*dx + dy*dy);

                if (dist < 50) {
                    score += items[i].points;
                    scoreEl.innerText = score;
                    items.splice(i, 1);
                } else if (items[i].y > canvas.height) {
                    items.splice(i, 1);
                }
            }
            frame++;
        }

        function draw() {
            ctx.clearRect(0, 0, canvas.width, canvas.height);
            
            // Draw Player
            ctx.font = "60px Arial";
            ctx.textAlign = "center";
            ctx.fillText(player.emoji, player.x, canvas.height - 60);

            // Draw Items
            ctx.font = "35px Arial";
            items.forEach(item => {
                ctx.fillText(item.emoji, item.x, item.y);
            });
        }

        function loop() {
            if (!gameRunning) return;
            update();
            draw();
            animationId = requestAnimationFrame(loop);
        }

        function startGame() {
            score = 0;
            frame = 0;
            items = [];
            scoreEl.innerText = "0";
            startScreen.classList.add('hidden');
            gameOverScreen.classList.add('hidden');
            gameRunning = true;
            loop();
            
            setTimeout(endGame, 30000); // 30 seconds game
        }

        function endGame() {
            gameRunning = false;
            cancelAnimationFrame(animationId);
            finalScoreEl.innerText = score;
            
            if(score > 200) feedbackEl.innerText = "أنتِ بطلة البتراء الحقيقية! فخورون بكِ.";
            else feedbackEl.innerText = "محاولة رائعة! مدرستنا تناديكِ للمزيد من العطاء.";
            
            gameOverScreen.classList.remove('hidden');
        }

        function shareScore() {
            const text = `حققت ${score} نقطة في تحدي انتماء مدرسة البتراء! 🏫💚`;
            navigator.clipboard.writeText(text);
            alert("تم نسخ النتيجة! انشريها الآن في تعليق على صفحة المدرسة.");
        }
